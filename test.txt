class Cake {
    init(flavour) {
        this.flavour = flavour;
    }

    taste() {
        var adjective = "delicious";
        print "The " + this.flavour + " cake is " + adjective + "!";
    }
}


var cake = Cake("carrot");
cake.taste();

print cake.flavour;
cake.flavour = "almond";
cake.taste();
